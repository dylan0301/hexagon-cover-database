\section{Exact and interval certificates}
\label{app:certificates}

This appendix records the finite certificates used above.  A certificate is
part of the proof only through the exact statement and verification argument
given here.  The listed programs are reproducibility aids; neither a plot nor
an unrecorded floating-point convention is used.

\subsection{The finite caliper certificate}

Let $R$ and $J$ denote counterclockwise rotations through $2\pi/3$ and
$\pi/2$, respectively.  For a nonempty finite set $S\subset\mathbb R^2$ and
a vector $N$, put
\begin{equation}
 W_S(N)=\sum_{k=0}^2\max_{z\in S}\langle z,R^kN\rangle .
 \label{eq:caliper-support-sum}
\end{equation}

\begin{theorem}[Finite caliper certificate]
\label{thm:cert-caliper}
If $S$ has at least two distinct points, then $S$ is contained in a closed
unit equilateral triangle if and only if there are distinct $p,q\in S$ and
$\varepsilon\in\{-1,1\}$ for which
\begin{equation}
 W_S\bigl(\varepsilon J(q-p)\bigr)
 \le \frac{\sqrt3}{2}\lVert p-q\rVert .
 \label{eq:finite-caliper-test}
\end{equation}
If $S$ consists of one point, it is contained in equilateral triangles of
arbitrarily small positive side length.

For $S_x=\{O,A,B,x\}$, every clause of
\eqref{eq:finite-caliper-test} is a finite polynomial clause of degree at
most four in the coordinates of $A,B,x$; at most $12\cdot4^3$ clauses are
required, and repeated point labels merely delete zero-length pairs.
\end{theorem}

\begin{proof}
For a unit vector $n$, write
\[
 G_S(n)=\sum_{k=0}^2h_S(R^kn),
 \qquad h_S(n)=\max_{z\in S}\langle z,n\rangle .
\]
The three supporting half-planes with normals $n,Rn,R^2n$ form an
equilateral triangle of side $2G_S(n)/\sqrt3$.  Hence $S$ fits in a closed
unit equilateral triangle precisely when
$\min_{\lVert n\rVert=1}G_S(n)\le\sqrt3/2$.

On an open normal-direction cell on which the three support points are
fixed, $G_S(\theta)=c\cos\theta+d\sin\theta$ and
$G_S''=-G_S<0$ unless all points coincide.  Thus a minimum occurs at a
cell boundary, where some support line contains two distinct points $p,q$.
After a cyclic rotation of the three normals, its normal is
$\varepsilon J(q-p)/\lVert p-q\rVert$.  Homogeneity yields
\eqref{eq:finite-caliper-test}.  A nondegenerate segment is checked directly
with its two normals.

For the polynomial expansion fix $(p,q,\varepsilon)$, put
$N=\varepsilon J(q-p)$ and $N_k=R^kN$, and choose support labels
$z_0,z_1,z_2\in S_x$.  The support cell is
\begin{equation}
 \langle z_k,N_k\rangle\ge\langle z,N_k\rangle
 \quad(z\in S_x,\ k=0,1,2),
 \label{eq:caliper-cell}
\end{equation}
with $\lVert p-q\rVert^2>0$.  Since $W_S(N)\ge0$, the remaining test is
\begin{equation}
 4\left(\sum_{k=0}^2\langle z_k,N_k\rangle\right)^2
 \le3\lVert p-q\rVert^2.
 \label{eq:caliper-cell-polynomial}
\end{equation}
The points are affine-linear in the parameters, so
\eqref{eq:caliper-cell}--\eqref{eq:caliper-cell-polynomial} have degree at
most four.  Six unordered pairs, two signs, and $4^3$ support-label triples
give the asserted finite exhaustive list.
\end{proof}

\subsection{The outward-rounded mixed-overlap certificate}

\begin{theorem}[Mixed cap-overlap certificate]
\label{thm:cert-mixed-overlap}
In the strict residual-core domain
\[
 0<a,b<1,\qquad a+b>1,\qquad a^2+ab+b^2<1,
\]
let $c_*$, $A$, $C$, $\eta$, $\Delta_0$, $u$, and the quantities
$r_X^0,\ell_X^0,P_X^0$ for $X\in\{A,C\}$ be those of
Section~\ref{sec:ab-core} and equations
\eqref{eq:mixed-certificate-quantities} below.  Then
\begin{equation}
 r_A^0\ge0,\qquad r_C^0\ge0,\qquad
 \Delta_0\le0,\qquad P_A^0\ge0,\qquad P_C^0\ge0.
 \label{eq:mixed-five-signs}
\end{equation}
Both $P$ inequalities hold at every fixed asymmetric parameter; reflection
is used only to restrict the domain to $a\ge b$.
\end{theorem}

\begin{proof}[Certificate verification]
Represent a global vector as $(x,(\sqrt3/2)y)$ and put $C'=R^{-1}C$.  For
$A=(x_A,(\sqrt3/2)y_A)$ and
$C'=(x',(\sqrt3/2)y')$, define
\begin{align}
 \Delta_0&=x_Ay'-y_Ax',&u&=\langle A,C'\rangle,\notag\\
 r_X^0&=\lVert X\rVert^2-\frac34(1-c_*)^2,
 &\ell_X^0&=(1-c_*)u-(2c_*-1)\lVert X\rVert^2,\notag\\
 P_X^0&=r_X^0\Delta_0^2-(\ell_X^0)^2.
 \label{eq:mixed-certificate-quantities}
\end{align}

By reflection take $z=a-b\ge0$.  Introduce all variables used by the
pruning test:
\begin{equation}
 \begin{aligned}
 p&=1-b,& q&=1-a,& s&=p+q,\\
 U&=1-s=a+b-1,& m&=q,& z&=p-q,
 \end{aligned}
 \label{eq:mixed-pruning-variables}
\end{equation}
and
\begin{equation}
 D^2=z^2+6U+3U^2=4(a^2+ab+b^2)-3.
 \label{eq:mixed-pruning-D2}
\end{equation}
Thus $p=(s+z)/2$, $q=(s-z)/2=m$, and the inverse conversion is
\begin{equation}
 a=1-m,\qquad b=1-(s-m).                        \label{eq:mixed-inverse-ab}
\end{equation}

The selected radial value has two complete rational charts.  On the $L$
sheet put $x=1+t$ and
\[
 c_*=\frac{x^2+3}{4x},\qquad
 m=\frac{(x^2+3)(x-1)(x+3)}{16x^2},\qquad
 s=c_*-v.
\]
Equivalently,
\begin{equation}
 z=s-2m,\qquad p=s-m,\qquad
 a=1-m,\qquad b=1-s+m.                         \label{eq:L-chart-ab}
\end{equation}
The identity $x=2c_*-\sqrt{4c_*^2-3}$ and the displayed formula for $m$
are exactly the selected-root equation
$c_*^4-c_*^2+mc_*-m^2=0$.  The actual reflected $L$-sheet domain is the
subset satisfying
\[
 z\ge0,\quad 0<m\le p<1,\quad U>0,\quad 0<D^2<1,\quad
 c_*\ge s,\quad \chi=s^4-s^2+mp\le0.
\]
It is enclosed by the closed rational superrectangle
\[
 0\le t\le\frac34,\qquad 0\le v\le\frac4{25}.
\]
Indeed $0\le t\le\sqrt3-1<3/4$ and
$0\le v\le U<2/\sqrt3-1<4/25$.

On the other sheet put $x=1+t$ and
\[
 s=\frac{x^2+3}{4x},\qquad
 E=\frac{3-x^2}{2x},\qquad
 c_*=s-v,\qquad z=sE-v(1+E).
\]
The exact inverse conversion is
\begin{equation}
 m=\frac{s-z}{2},\qquad p=\frac{s+z}{2},\qquad
 a=1-\frac{s-z}{2},\qquad b=1-\frac{s+z}{2}.   \label{eq:T-chart-ab}
\end{equation}
Here $x=2s-\sqrt{4s^2-3}$ and $c_*=(s+z)/(1+E)$, so $v\ge0$ is the
selected side of the sheet, with $v=0$ its selector face.  Its actual
reflected domain is the subset satisfying
\[
 z\ge0,\quad 0<m\le p<1,\quad U>0,\quad 0<D^2<1,\quad
 c_*>\frac23,\quad \chi=s^4-s^2+mp\ge0,
\]
and it is enclosed by
\[
 0\le t\le\frac34,\qquad 0\le v\le\frac13.
\]
The bounds follow from $x\le\sqrt3<7/4$ and $0\le v<1/3$.
These closed superrectangles include all selector and endpoint faces.  They
also contain points outside the original domain; proving the certificate
there only strengthens the enclosure.

Every binary64 endpoint operation is widened by one next-representable
number toward the relevant infinity.  Rational constants are enclosed from
both sides.  Square-root endpoints are moved until exact integer-ratio
comparisons prove that their squares enclose the radicand.  Thus the square
root step has no unverified library-rounding premise.

A box is pruned only when outward intervals prove that it cannot satisfy a
necessary domain condition
\[
 x^2\le3,\quad z\ge0,\quad m\ge0,\quad U\ge0,\quad0\le D^2\le1,
\]
and, on the second sheet, $c_*>2/3$.  Before certification, the entire box
must lie in $D^2>0$ and the two Newton denominators must have negative upper
endpoints.  For each target $f$, centered first-order interval automatic
differentiation uses
\[
 f(B)\subseteq f(t_0,v_0)+\partial_t f(B)(t-t_0)
                         +\partial_v f(B)(v-v_0).
\]
Unresolved boxes are bisected along the larger relative chart width through
depth $16$.

On the $L$ chart the run processes $21040$ boxes, prunes $1258$, certifies
$9907$, and leaves none unresolved; the least certified lower bound for a
mixed $P$ is
\[
 2.522678787629918\cdot10^{-7}.
\]
On the other chart it
processes $18574$, prunes $222$, certifies $9208$, and leaves none
unresolved; the corresponding least bound is
\[
 9.92316118814129\cdot10^{-7}.
\]
The deterministic output reports
\texttt{result: PASS} and \texttt{total\_unresolved\_boxes: 0}.

The reproduction command is
\begin{verbatim}
python3 proof/3XXX_CE0/31XX_Nplus1/310X_all_Vd0/
  3103X_residual_core/3103X_computation/
  verify_tangent_witness_mixed_overlap.py
\end{verbatim}
with line breaks removed.  Directed enclosure of every retained box and
exhaustive bisection of both superrectangles prove
\eqref{eq:mixed-five-signs} on the full parameter domain.
\end{proof}
